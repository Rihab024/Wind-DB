💻 Requêtes SQL — Analyses clés
Requête 1 — Top 5 composants les plus coûteux
SELECT c.Name || '/' || c.Subcomponent AS Composant,
  COUNT(di.DefectInstanceID) AS NbIncidents,
  ROUND(SUM(di.RepairCost_EUR), 0) AS CoutTotal_EUR
FROM Defect_Instances di
JOIN Components c ON di.ComponentID = c.ComponentID
GROUP BY c.ComponentID
ORDER BY CoutTotal_EUR DESC LIMIT 5;
Requête 2 — Matériaux par coût moyen
SELECT m.Name, m.Category,
  COUNT(di.DefectInstanceID) AS NbIncidents,
  ROUND(AVG(di.RepairCost_EUR), 0) AS CoutMoyen
FROM Materials m
JOIN Defect_Instances di ON m.MaterialID = di.MaterialID
GROUP BY m.MaterialID
ORDER BY CoutMoyen DESC LIMIT 5;
